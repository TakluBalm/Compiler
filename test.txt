<syntax> : <rule> | <rule> <syntax> ;

<rule> : <rule_name> ":" <definition> <line_end>;

<line_end> : ";" | ";" <line_end> ;

<definition> : <list> | <list> "|" <definition> ;

<list> : <term> | <term> <list> ;

<term> : <const> | <rule_name> ;

<rule_name> : "<" <name> ">";

<name> :	<character>
		|	<name> <character>;

<character> :	<letter> | <digit> | <symbol> ;

<letter> :	"A-Z" | "a-z";

<symbol> : "_" | "-";

<digit> :	<octal_digit> | "8" | "9" ;

<char> : <character> | <special_char> ;

<const> : "\042" <string> "\042";

<string> :	<char> |
			<string> <char> ;

<special_char> :	"\134" <octal_digit> | 
					"\134" <octal_digit> <octal_digit> |
					"\134" <octal_digit> <octal_digit> <octal_digit> ;

<octal_digit> :	"0" | "1" | "2" | "3" | "4" | "5" | "6" | "7" ;
